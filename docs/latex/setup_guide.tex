\section{Prerequisites}
\begin{itemize}
  \item Docker Engine 24+ and Docker Compose v2
  \item Git
  \item (Optional) Ollama for local AI
\end{itemize}

\section{Quick Start}
\begin{lstlisting}[language=bash]
git clone https://github.com/Personfu/CyberDeck.git
cd CyberDeck/infra/docker
docker compose up --build
\end{lstlisting}
\begin{itemize}
  \item Web UI: \url{http://localhost:3000}
  \item API: \url{http://localhost:8000}
  \item Health: \url{http://localhost:8000/healthz}
\end{itemize}

\section{Raspberry Pi 400 Deployment}
\begin{lstlisting}[language=bash]
# On Pi 400 running Kali ARM or Raspberry Pi OS
sudo ./scripts/requirements_setup.sh
sudo ./scripts/lcd_install.sh     # DFRobot 3.5" TFT
sudo reboot
sudo ./scripts/tplink_patch.sh    # TL-WN722N WiFi
sudo reboot

# Start CyberDeck platform
cd CyberDeck/infra/docker
docker compose up --build -d
\end{lstlisting}

\subsection{Display Config}
In \texttt{/boot/config.txt}:
\begin{lstlisting}
dtoverlay=waveshare35a,rotate=270,invertx=1,swapxy=1
\end{lstlisting}

\section{Building LaTeX Documentation}
\begin{lstlisting}[language=bash]
cd docs/latex
# Requires texlive-latex-base, texlive-latex-extra
for f in *.tex; do latexmk -pdf "$f"; done
\end{lstlisting}

\section{GitHub Actions}
The repository includes two workflows:
\begin{description}
  \item[ci.yml] Runs on push/PR --- API tests, LaTeX build, Docker build, Compose smoke test
  \item[release.yml] Runs on tags --- publishes images to GHCR, attaches PDFs to release
\end{description}
